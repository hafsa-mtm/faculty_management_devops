pipeline {
    agent any
environment {
        SONARCLOUD_URL = 'https://sonarcloud.io' // URL de SonarCloud
		SONARCLOUD_ORGANIZATION = 'discoveryorg' // Nom de votre organisation SonarCloud
        SONARCLOUD_TOKEN = credentials('SonarCloud_token')// Token généré dans votre profil SonarCloud
         SONARCLOUD_PROJECT_KEY = 'discoveryorg_discovery-service' // full project key

        
        registryCredential = 'metmari'
        dockerimagename = "faculty_management-faculty-discovery-service:latest"
        
    }
    
    stages {
        stage('Clone Repository') {
            steps {
                git branch: 'main', 
                    url: 'https://github.com/hafsa-mtm/faculty_management_devops.git',
                    credentialsId: 'Github_token'
            }
        }
        stage('Compile') {
            steps {
                dir('Discovery-Service') { 
                    bat 'mvn clean compile'
                }
            }
        }
        
           stage('Test Unitaire') {
            steps {
                dir('Discovery-Service') { 
                    bat 'mvn test'
                }
            }
        }
        
        
stage('Code Quality') {
    steps {
        dir('Discovery-Service') { 
            withEnv(["SONAR_TOKEN=${SONARCLOUD_TOKEN}"]) {
                bat """
                mvn org.sonarsource.scanner.maven:sonar-maven-plugin:sonar ^
                    -Dsonar.projectKey=${SONARCLOUD_PROJECT_KEY} ^
                    -Dsonar.organization=${SONARCLOUD_ORGANIZATION} ^
                    -Dsonar.host.url=${SONARCLOUD_URL} ^
                    -Dsonar.login=${SONARCLOUD_TOKEN} ^
                    -Djava.net.preferIPv4Stack=true
                """
            }
        }
    }
}


        
           stage('Package Jar') {
            steps {
                dir('Discovery-Service') { 
                    bat 'mvn clean package -DskipTests'
                }
            }
        }
        
       stage('Build image') {
    steps {
        dir('Discovery-Service') {
            script {
                def dockerImage = docker.build(env.dockerimagename)
            }
        }
    }
}

        /*
         stage('Pushing Image') {
          steps{
            dir('Discovery-Service') {    
            script {
              docker.withRegistry( 'https://registry.hub.docker.com', registryCredential ) {
                dockerImage.push("latest")
              }
            }
            }
          }
        }
        */
        
        stage('Deploying App to Kubernetes') {
          steps {
            dir('Discovery-Service') {  
            script {
              kubernetesDeploy(configs: "Discovery-Deploy.yaml", kubeconfigId: "Kubernetes")
            }
          }
          }
        }
        
        
    }

	post {
        always {
            echo "Build complete. Sending notification..."
        }
        success {
            mail to: 'hafsamitmari@gmail.com',
                 subject: "Discovery-Service Build #${env.BUILD_NUMBER} Success",
                 body: "The build succeeded!"
        }
        failure {	
            mail to: 'hafsamitmari@gmail.com',
                 subject: "Build #${env.BUILD_NUMBER} Failed",
                 body: "The build failed. Please check Jenkins logs."
        }
    }
    
}
